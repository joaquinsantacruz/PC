Monitor Mesa
    ColaEspecial cola
    cond turno[N], llego, salgo

    procedure llegada(id: in int, prioridad: in boolean)
        cola.push(id, prioridad)
        signal(llegue)
        wait(turno[id])
    

    procedure salida() {
        signal(salgo)
    }


    procedure siguiente() {
        if (cola.isEmpty())
            wait(llego)
        signal(turno[cola.pop()])
        wait(salgo)
    }


process Persona[id: 0..N-1]
    boolean prioridad = getPrioridad()
    Mesa.llegada(id, prioridad)
    Votar()
    Mesa.salida()


process Autoridad
    for i = 0 to N - 1
        Mesa.siguiente()
